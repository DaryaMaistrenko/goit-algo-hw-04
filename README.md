### 1. –°–æ—Ä—Ç—É–≤–∞–Ω–Ω—è –∑–ª–∏—Ç—Ç—è–º ###

def merge_sort(arr):
    if len(arr) > 1:
        mid = len(arr) // 2  # –ó–Ω–∞–π–¥–µ–º–æ —Å–µ—Ä–µ–¥–∏–Ω—É –º–∞—Å–∏–≤—É
        left_half = arr[:mid]  # –õ—ñ–≤–∞ –ø–æ–ª–æ–≤–∏–Ω–∞
        right_half = arr[mid:]  # –ü—Ä–∞–≤–∞ –ø–æ–ª–æ–≤–∏–Ω–∞

        merge_sort(left_half)  # –†–µ–∫—É—Ä—Å–∏–≤–Ω–æ —Å–æ—Ä—Ç—É—é—á–∏ –ª—ñ–≤—É –ø–æ–ª–æ–≤–∏–Ω—É
        merge_sort(right_half)  # –†–µ–∫—É—Ä—Å–∏–≤–Ω–æ —Å–æ—Ä—Ç—É—é—á–∏ –ø—Ä–∞–≤—É –ø–æ–ª–æ–≤–∏–Ω—É

        i = j = k = 0

        # –û–±'—î–¥–Ω–∞–Ω–Ω—è –¥–≤–æ—Ö –ø–æ–ª–æ–≤–∏–Ω
        while i < len(left_half) and j < len(right_half):
            if left_half[i] < right_half[j]:
                arr[k] = left_half[i]
                i += 1
            else:
                arr[k] = right_half[j]
                j += 1
            k += 1

        # –ü–µ—Ä–µ–≤—ñ—Ä—è—î–º–æ, —á–∏ –∑–∞–ª–∏—à–∏–ª–∏—Å—è –µ–ª–µ–º–µ–Ω—Ç–∏ –≤ –ª—ñ–≤—ñ–π –ø–æ–ª–æ–≤–∏–Ω—ñ
        while i < len(left_half):
            arr[k] = left_half[i]
            i += 1
            k += 1

        # –ü–µ—Ä–µ–≤—ñ—Ä—è—î–º–æ, —á–∏ –∑–∞–ª–∏—à–∏–ª–∏—Å—è –µ–ª–µ–º–µ–Ω—Ç–∏ –≤ –ø—Ä–∞–≤—ñ–π –ø–æ–ª–æ–≤–∏–Ω—ñ
        while j < len(right_half):
            arr[k] = right_half[j]
            j += 1
            k += 1


### 2. –°–æ—Ä—Ç—É–≤–∞–Ω–Ω—è –≤—Å—Ç–∞–≤–∫–∞–º–∏ ### 

def insertion_sort(arr):
    for i in range(1, len(arr)):
        key = arr[i]
        j = i - 1
        while j >= 0 and key < arr[j]:
            arr[j + 1] = arr[j]
            j -= 1
        arr[j + 1] = key


### 3. Timsort (–≤–±—É–¥–æ–≤–∞–Ω–∏–π –≤ Python) ### 

def timsort(arr):
    return sorted(arr)
